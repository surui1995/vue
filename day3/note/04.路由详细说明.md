路由的使用方式
在项目中的router文件夹里面的index.js
先引入：
```
import Vue from 'vue'
import Router from 'vue-router'
Vue.use(Router) //使用路由
```
```
import Hello from '@/components/Hello' //通过引入外部模板的方式
const Study = {template:`<div>我是学习的页面</div>`} //在页面直接声明的方式
export default new Router({
  routes: [
    {
      path: '/',  //规定路径访问的路劲径
      name: 'Hello',
      component: Hello
    },{
      path: '/Study',  自定义路径
      name: 'Study',
      component: Study
    }]
})
    ```
